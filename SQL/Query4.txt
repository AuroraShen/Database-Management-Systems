SELECT s.S_NAME AS [SupplierName], COUNT(DISTINCT o.O_CUSTKEY) AS CustCount
FROM LINEITEM AS l
LEFT OUTER JOIN SUPPLIER AS s
ON l.L_SUPPKEY = s.S_SUPPKEY
LEFT OUTER JOIN ORDERS AS o
ON l.L_ORDERKEY = o.O_ORDERKEY
GROUP BY SupplierName
HAVING CustCount > 615
ORDER BY CustCount DESC;
